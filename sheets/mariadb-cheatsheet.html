mysql CLI — Verbindung & Grundbefehle

DDL — Datenbanken & Tabellen
Datenbanken
CREATE DATABASE nova CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci; DROP DATABASE IF EXISTS nova; SHOW CREATE DATABASE nova;
Tabellen erstellen & ändern
CREATE TABLE servers ( id INT NOT NULL AUTO_INCREMENT, uuid VARCHAR(36) UNIQUE NOT NULL, name VARCHAR(255), status ENUM('active','error') DEFAULT 'active', created DATETIME DEFAULT CURRENT_TIMESTAMP, PRIMARY KEY (id), INDEX idx_uuid (uuid) ) ENGINE=InnoDB; ALTER TABLE servers ADD COLUMN deleted TINYINT DEFAULT 0; ALTER TABLE servers ADD INDEX (status); ALTER TABLE servers DROP COLUMN old_col; RENAME TABLE old_name TO new_name;

Benutzerverwaltung & Berechtigungen
Benutzer anlegen & löschen
CREATE USER 'nova'@'%' IDENTIFIED BY 'secret'; CREATE USER 'nova'@'10.0.0.%' IDENTIFIED BY 'secret'; ALTER USER 'nova'@'%' IDENTIFIED BY 'newpass'; DROP USER 'nova'@'%'; SHOW GRANTS FOR 'nova'@'%';
Berechtigungen vergeben
GRANT ALL PRIVILEGES ON nova.* TO 'nova'@'%'; GRANT SELECT, INSERT, UPDATE ON db.* TO 'user'@'%'; GRANT REPLICATION SLAVE ON *.* TO 'replica'@'%'; REVOKE ALL ON nova.* FROM 'nova'@'%'; FLUSH PRIVILEGES;

Backup & Restore — mysqldump
Backup erstellen
# Einzelne Datenbank mysqldump -u root -p nova > nova_backup.sql # Mit Struktur-DDL und Daten mysqldump --single-transaction --routines \ --triggers nova > nova.sql # Alle Datenbanken mysqldump --all-databases --single-transaction > all.sql # Nur Struktur mysqldump --no-data nova > schema.sql # Komprimiert mysqldump nova | gzip > nova.sql.gz
Restore
mysql -u root -p nova < nova_backup.sql zcat nova.sql.gz | mysql -u root -p nova
💡 --single-transaction sichert InnoDB ohne table locks (MVCC)

Replikation — Master/Slave (klassisch)
Master konfigurieren
# /etc/mysql/my.cnf [mysqld] server-id = 1 log_bin = /var/log/mysql/mysql-bin.log binlog_do_db = nova SHOW MASTER STATUS; # File + Position merken
Slave konfigurieren
CHANGE MASTER TO MASTER_HOST='10.0.0.1', MASTER_USER='replica', MASTER_PASSWORD='secret', MASTER_LOG_FILE='mysql-bin.000001', MASTER_LOG_POS=154; START SLAVE; SHOW SLAVE STATUS\G # Seconds_Behind_Master prüfen STOP SLAVE; RESET SLAVE;

Galera Cluster (MariaDB Cluster)
Cluster-Status
SHOW STATUS LIKE 'wsrep%'; SHOW STATUS LIKE 'wsrep_cluster_size'; SHOW STATUS LIKE 'wsrep_local_state_comment'; # Synced = OK, Joined = in sync
galera.cnf — Schlüsseloptionen
[galera] wsrep_on = ON wsrep_provider = /usr/lib/galera/libgalera_smm.so wsrep_cluster_name = "openstack_cluster" wsrep_cluster_address = "gcomm://10.0.0.1,10.0.0.2,10.0.0.3" wsrep_sst_method = rsync binlog_format = ROW
Cluster starten (nach Totalausfall)
# Auf dem letzten aktiven Node: galera_new_cluster # oder: mysqld_safe --wsrep-new-cluster &

EXPLAIN & Query-Analyse
EXPLAIN verwenden
EXPLAIN SELECT * FROM instances WHERE uuid='...'; EXPLAIN FORMAT=JSON SELECT ...; EXPLAIN EXTENDED SELECT ...; # + SHOW WARNINGS
EXPLAIN-Ausgabe verstehen
type: ALL(schlecht) → index → range → ref → eq_ref → const(gut) key: verwendeter Index (NULL = kein Index) rows: geschätzte Zeilenanzahl Extra: Using filesort / Using temporary = ACHTUNG
Slow Query Log
SET GLOBAL slow_query_log = 'ON'; SET GLOBAL long_query_time = 1; SET GLOBAL slow_query_log_file = '/var/log/mysql/slow.log'; mysqldumpslow -t 10 /var/log/mysql/slow.log

my.cnf — Wichtige Konfigurationsoptionen
Performance & InnoDB
[mysqld] innodb_buffer_pool_size = 4G # 60-80% RAM innodb_log_file_size = 256M innodb_flush_log_at_trx_commit = 2 # 1=sicher, 2=schneller innodb_flush_method = O_DIRECT max_connections = 500 thread_cache_size = 100 query_cache_size = 0 # deaktivieren tmp_table_size = 64M max_heap_table_size = 64M open_files_limit = 65535 character-set-server = utf8mb4 collation-server = utf8mb4_general_ci
💡 OpenStack-DBs: nova, neutron, cinder, keystone, heat — alle brauchen UTF8MB4
