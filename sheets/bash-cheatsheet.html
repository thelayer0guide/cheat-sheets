Robustes Skript-Template
#!/usr/bin/env bash set -euo pipefail # exit on error IFS=$'\n\t' readonly SCRIPT_DIR=$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd) usage() { echo "Usage: $0 [-v] <arg>" exit 1 } cleanup() { echo "Cleanup..." >&2 } trap cleanup EXIT

while getopts ":hvo:" opt; do case $opt in h) usage ;; v) VERBOSE=1 ;; o) OUTPUT="$OPTARG" ;; ?) echo "Ungültig: -$OPTARG"; exit 1 ;; esac done shift $(( OPTIND - 1 ))
