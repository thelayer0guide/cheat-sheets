Setup & Init
Konfiguration
git config --global user.name "Name" git config --global user.email "mail" git config --global core.editor vim git config --global init.defaultBranch main git config --list
Repository erstellen
git init git init my-project git clone https://github.com/org/repo.git git clone repo.git --depth 1 # shallow git clone repo.git -b develop
📦
Staging & Commits
Status & Staging
git status git status -s # kurz git add datei.txt git add . # alles git add -p # interaktiv (hunks) git restore --staged file # unstage git restore file # Änderungen verwerfen
Committen
git commit -m "feat: add login" git commit -am "fix: ..." # add + commit git commit --amend # letzten ändern git commit --amend --no-edit # nur Dateien
Diff
git diff # unstaged git diff --staged # staged git diff main..feature # Branch-Diff git diff HEAD~2 HEAD # letzte 2 git diff --stat # nur Statistik
💡 git add -p für saubere, atomare Commits — jeder Hunk einzeln prüfen
🌿
Branches
Branch verwalten
git branch # lokal auflisten git branch -a # inkl. remote git branch feature/login # erstellen git branch -d feature # löschen (safe) git branch -D feature # force-delete git branch -m old new # umbenennen
Wechseln & Erstellen
git switch main git switch -c feature/neu # erstellen + wechseln git switch - # vorheriger Branch # alt (checkout) git checkout -b feature
Merge
git merge feature git merge --no-ff feature # merge commit git merge --squash feature # squash git merge --abort # abbrechen git mergetool # Konflikte lösen
Rebase
git rebase main git rebase -i HEAD~3 # interaktiv git rebase --onto main old new git rebase --abort git rebase --continue
💡 Interaktiver Rebase: pick squash fixup reword drop zum Aufräumen der History
🔄
Remote
Remote verwalten
git remote -v git remote add origin url git remote rename origin upstream git remote remove origin git remote set-url origin neue-url
Push & Pull
git push origin main git push -u origin feature # tracking setzen git push --force-with-lease # safe force push git push origin :branch # remote löschen git push --tags git pull # fetch + merge git pull --rebase # fetch + rebase git fetch origin git fetch --prune # tote Branches weg
🏷
Tags
Tags erstellen & pushen
git tag v1.0.0 git tag -a v1.0.0 -m "Release 1.0" git tag -a v1.0.0 abc1234 # auf Commit git tag # auflisten git push origin v1.0.0 git push origin --tags git tag -d v1.0.0 # lokal löschen git push origin --delete v1.0.0
📜
History & Log
Log
git log --oneline git log --oneline --graph --all git log -p # mit Diffs git log --author="Name" git log --since="2024-01-01" git log --grep="fix" git log -S "Funktionsname" # pickaxe git log -- pfad/datei.txt # Datei-History
Suchen & Blame
git blame datei.txt git blame -L 10,20 datei # Zeilen 10-20 git grep "muster" git show abc1234 git show HEAD~2:datei.txt
🧰
Stash, Reset & Restore
Stash
git stash git stash push -m "wip: feature" git stash list git stash pop # anwenden + löschen git stash apply stash@{1} # anwenden git stash drop stash@{0} git stash clear git stash branch neu # → Branch
Reset
git reset --soft HEAD~1 # commit rückgängig, staged bleibt git reset --mixed HEAD~1 # + unstage (default) git reset --hard HEAD~1 # ⚠ Änderungen weg! git reset HEAD datei # unstage
Cherry-Pick & Revert
git cherry-pick abc1234 git cherry-pick A..B # Bereich git revert abc1234 # sicheres Rückgängig git revert HEAD~3..HEAD # mehrere
💡 git reflog — rettet verlorene Commits nach reset --hard oder Branch-Löschung

.gitconfig & Aliases
Nützliche Aliases

git worktree add ../hotfix hotfix/bug git worktree list git worktree remove ../hotfix
Submodules
git submodule add url pfad git submodule update --init --recursive git clone --recurse-submodules url git submodule foreach 'git pull'
Maintenance
