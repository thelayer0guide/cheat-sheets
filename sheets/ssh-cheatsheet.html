Verbinden & Grundbefehle
Verbindung aufbauen
ssh user@host ssh -p 2222 user@host ssh -i ~/.ssh/mykey user@host ssh -o StrictHostKeyChecking=no user@host ssh -v user@host # verbose/-vvv debug
Befehl remote ausführen
ssh user@host 'uptime' ssh user@host 'sudo systemctl status nginx' ssh user@host 'bash -s' < local_script.sh
Dateien übertragen
scp file.txt user@host:/remote/path/ scp -r dir/ user@host:/remote/ scp -P 2222 user@host:/remote/file . rsync -avz -e ssh dir/ user@host:/dest/
🗝
Key-Management
Schlüssel generieren
ssh-keygen -t ed25519 -C "user@host" ssh-keygen -t rsa -b 4096 -C "deploy key" ssh-keygen -t ed25519 -f ~/.ssh/mykey -N "" # Fingerprint anzeigen: ssh-keygen -lf ~/.ssh/id_ed25519.pub
Key deployen
ssh-copy-id -i ~/.ssh/id_ed25519.pub user@host # Manuell: cat ~/.ssh/id_ed25519.pub | ssh user@host \ 'mkdir -p ~/.ssh && cat >> ~/.ssh/authorized_keys'
ssh-agent
eval $(ssh-agent -s) ssh-add ~/.ssh/id_ed25519 ssh-add -l # Liste geladener Keys ssh-add -d ~/.ssh/id_ed25519 # Key entfernen ssh -A user@host # Agent-Forwarding
⚙
~/.ssh/config
Host-Konfiguration
Host bastion HostName 1.2.3.4 User admin IdentityFile ~/.ssh/id_ed25519 Port 2222 Host prod-* User deploy ProxyJump bastion IdentityFile ~/.ssh/prod_key Host * ServerAliveInterval 60 ServerAliveCountMax 3 AddKeysToAgent yes
ProxyJump (Jump Host)
# Über Bastion zu internem Host: ssh -J user@bastion user@internal-host # Kette von Jump-Hosts: ssh -J bastion1,bastion2 user@target
🌐
Tunneling & Port Forwarding
Local Port Forward (-L)
# Zugriff auf Remote-DB über local:5432: ssh -L 5432:db-internal:5432 user@bastion # Im Hintergrund: ssh -fNL 8080:internal:80 user@bastion
Remote Port Forward (-R)
# Remote-Host exponiert local:8080 nach außen: ssh -R 8080:localhost:80 user@server # Für alle Interfaces auf remote: ssh -R 0.0.0.0:8080:localhost:80 user@server
SOCKS-Proxy (-D)
# Lokaler SOCKS5-Proxy über SSH: ssh -D 1080 -fN user@server # Browser: SOCKS5 Proxy → 127.0.0.1:1080
SFTP
sftp user@host sftp> ls / cd / get / put / rm sftp> get /remote/file.txt ./local/ sftp> put local.txt /remote/path/ sftp> mget *.log # mehrere Dateien

sshd — Server-Konfiguration
/etc/ssh/sshd_config (Wichtige Optionen)
Port 22 PermitRootLogin no PasswordAuthentication no PubkeyAuthentication yes AuthorizedKeysFile .ssh/authorized_keys AllowUsers user1 user2 AllowGroups sshusers X11Forwarding no MaxAuthTries 3 LoginGraceTime 30 ClientAliveInterval 60
sshd verwalten
sshd -t # Config testen systemctl reload sshd ss -tlnp | grep sshd # Port prüfen last # Login-History journalctl -u sshd -f # Logs

Zertifikate & Known Hosts
CA-basierte Authentifizierung
# CA-Key erstellen: ssh-keygen -t ed25519 -f ssh_ca # User-Key signieren: ssh-keygen -s ssh_ca -I "user@org" \ -n username -V +52w id_ed25519.pub # In sshd_config: TrustedUserCAKeys /etc/ssh/ssh_ca.pub
Known Hosts
# Host-Key entfernen: ssh-keygen -R hostname # Host-Key vorab eintragen: ssh-keyscan -H hostname >> ~/.ssh/known_hosts # Host fingerprint prüfen: ssh-keyscan hostname | ssh-keygen -lf -
