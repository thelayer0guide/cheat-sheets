Basics
Variablen & Typen
local name = "Alice" -- string local count = 42 -- number (float oder int) local flag = true -- boolean local nothing = nil -- nil -- Globale Variable (ohne local): global_var = "sichtbar überall" print(type(name)) -- "string"
Strings
local s = "Hello, " .. name -- Verkettung mit .. local len = #s -- Länge string.upper(s) string.find(s, "pattern") string.format("%d: %s", count, name) string.sub(s, 1, 5) -- Substring (1-basiert!) string.gsub(s, "old", "new")
Kontrollfluss
if x > 0 then print("positiv") elseif x == 0 then print("null") else print("negativ") end

Tables (Kernstruktur)
Array & Hash
-- Array (1-basiert!): local arr = {"a", "b", "c"} print(arr[1]) -- "a" print(#arr) -- 3 table.insert(arr, "d") table.remove(arr, 1) -- Dictionary: local person = { name = "Alice", age = 30 } print(person.name) -- oder person["name"]
Iteration
-- Array (ipairs, geordnet): for i, v in ipairs(arr) do print(i, v) end -- Dictionary (pairs, ungeordnet): for k, v in pairs(person) do print(k, v) end
Schleifen
for i = 1, 10 do print(i) end for i = 10, 1, -1 do ... end -- Schritt -1 while condition do ... end repeat ... until condition
🔧
Funktionen
Definitionen
local function greet(name) return "Hello, " .. name end -- Mehrere Rückgabewerte: local function divmod(a, b) return math.floor(a/b), a % b end local q, r = divmod(10, 3) -- q=3, r=1
Varargs & Closures
local function sum(...) local total = 0 for _, v in ipairs({...}) do total = total + v end return total end -- Closure: local function counter() local n = 0 return function() n = n + 1; return n end end
🎯
Metatables & OOP
Klasse via Metatables
local Animal = {} Animal.__index = Animal function Animal.new(name, sound) return setmetatable({name=name, sound=sound}, Animal) end function Animal:speak() print(self.name .. ": " .. self.sound) end local cat = Animal.new("Cat", "meow") cat:speak() -- cat.speak(cat) implizit
Metatable Operatoren
setmetatable(t, { __index = parent, -- Vererbung / Fallback __newindex = fn, -- beim Setzen von Keys __tostring = fn, -- tostring(t) __add = fn, -- t1 + t2 __len = fn, -- #t __call = fn, -- t() })
🔀
Coroutines
Coroutine Basics
local co = coroutine.create(function(a, b) print("start", a, b) local c = coroutine.yield(1) print("resumed", c) end) coroutine.resume(co, 10, 20) -- start 10 20 coroutine.resume(co, 99) -- resumed 99 coroutine.status(co) -- "dead"
Module Pattern
-- mymod.lua: local M = {} function M.hello() print("hi") end return M -- Nutzung: local mymod = require("mymod") mymod.hello()
Tipp: Immer `local` verwenden. Globale Variablen verlangsamen Lookups und führen zu schwer debugbaren Problemen in eingebetteten Umgebungen (Neovim, Redis, nginx/OpenResty).
