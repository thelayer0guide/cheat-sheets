DDL — Schema
Tabelle erstellen
CREATE TABLE users ( id BIGSERIAL PRIMARY KEY, email TEXT NOT NULL UNIQUE, name TEXT, age INT CHECK (age > 0), role TEXT DEFAULT 'user', created TIMESTAMPTZ DEFAULT NOW() );
Tabelle ändern
ALTER TABLE users ADD COLUMN bio TEXT; ALTER TABLE users DROP COLUMN bio; ALTER TABLE users RENAME TO accounts; ALTER TABLE users ALTER COLUMN age SET NOT NULL;
Typen & Enums
CREATE TYPE status AS ENUM ('active','inactive'); -- Nützliche Typen: TEXT VARCHAR(n) INT BIGINT NUMERIC(p,s) BOOLEAN DATE TIMESTAMP TIMESTAMPTZ UUID JSONB ARRAY INET BYTEA
📊
DML — Daten
INSERT / UPSERT
INSERT INTO users (email, name) VALUES ('alice@x.de', 'Alice'), ('bob@x.de', 'Bob'); -- Upsert (conflict handling) INSERT INTO users (email, name) VALUES ('alice@x.de', 'Alice') ON CONFLICT (email) DO UPDATE SET name = EXCLUDED.name;
UPDATE / DELETE
UPDATE users SET role = 'admin' WHERE id = 1 RETURNING *; DELETE FROM users WHERE created < NOW() - '30 days'::INTERVAL;
SELECT Grundlagen
SELECT u.name, o.total FROM users u JOIN orders o ON o.user_id = u.id WHERE o.total > 100 ORDER BY o.total DESC LIMIT 10 OFFSET 20;
🔗
Joins & Aggregation
Join-Typen
INNER JOIN -- nur Matches LEFT JOIN -- alle links + Matches RIGHT JOIN -- alle rechts + Matches FULL JOIN -- alle + Matches CROSS JOIN -- kartesisches Produkt
GROUP BY & Aggregate
SELECT role, COUNT(*), AVG(age), SUM(orders), MAX(created) FROM users GROUP BY role HAVING COUNT(*) > 5;
Window Functions
SELECT name, salary, RANK() OVER (PARTITION BY dept ORDER BY salary DESC), LAG(salary) OVER (ORDER BY created) FROM employees;
CTEs (WITH)
WITH top_users AS ( SELECT user_id, COUNT(*) cnt FROM orders GROUP BY user_id ) SELECT u.name, t.cnt FROM users u JOIN top_users t ON u.id = t.user_id;

Indizes
Index erstellen
CREATE INDEX idx_users_email ON users (email); CREATE UNIQUE INDEX ON users (email); CREATE INDEX CONCURRENTLY ON orders (created_at); -- Partial Index: CREATE INDEX ON users (email) WHERE active = true; -- GIN für JSONB/Arrays: CREATE INDEX ON docs USING GIN (data);
EXPLAIN & Analyse
EXPLAIN SELECT * FROM users WHERE email = 'x'; EXPLAIN ANALYZE SELECT ... -- wirklich ausführen EXPLAIN (ANALYZE, BUFFERS, FORMAT JSON) ...

JSONB & Arrays
JSONB Operatoren
data->'key' -- JSON-Wert data->>'key' -- Text-Wert data #>> '{a,b}' -- verschachtelter Pfad data @> '{"role":"admin"}' -- enthält data ? 'key' -- key existiert jsonb_set(data, '{key}', '"val"')
Array-Operatoren
arr @> ARRAY[1,2] -- enthält arr && ARRAY[1,2] -- überschneidet array_append(arr, 5) unnest(arr) -- zu Zeilen expandieren array_agg(x) -- Zeilen zu Array

Backup & Administration
Backup & Restore
pg_dump -U postgres mydb > backup.sql pg_dump -Fc mydb > backup.dump # custom format pg_restore -U postgres -d mydb backup.dump psql -U postgres mydb < backup.sql
Rollen & Rechte
CREATE USER app WITH PASSWORD 'secret'; CREATE DATABASE mydb OWNER app; GRANT SELECT, INSERT ON ALL TABLES IN SCHEMA public TO app; GRANT USAGE ON ALL SEQUENCES IN SCHEMA public TO app;
Transaktionen
BEGIN; UPDATE accounts SET balance = balance - 100 WHERE id = 1; UPDATE accounts SET balance = balance + 100 WHERE id = 2; COMMIT; -- oder ROLLBACK; SAVEPOINT sp1; ... ROLLBACK TO sp1;
Tipp: EXPLAIN ANALYZE immer vor dem Deploy laufen lassen. Fehlende Indizes auf WHERE/JOIN-Spalten sind die häufigste Performance-Ursache.
