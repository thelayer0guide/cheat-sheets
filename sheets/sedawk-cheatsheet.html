sed — Grundsyntax & Flags
Syntax: sed [OPTIONS] 'ADRESSE BEFEHL' DATEI
sed 's/alt/neu/g' datei.txt sed -n '5p' datei.txt # nur Zeile 5 ausgeben sed -i 's/foo/bar/g' datei.txt # in-place bearbeiten sed -i.bak 's/foo/bar/' datei.txt# mit Backup sed -E 's/[0-9]+/NUM/g' datei.txt # extended regex
Flags im s-Befehl
s/alt/neu/g # global (alle Vorkommen) s/alt/neu/I # case-insensitive s/alt/neu/2 # nur 2. Vorkommen s/alt/neu/p # mit -n: Treffer ausgeben s/alt/neu/gI # global + case-insensitive
📍
sed — Adressen (Zeilenbereiche)
Zeilen- und Muster-Adressen
sed '3s/foo/bar/' f # nur Zeile 3 sed '2,5s/foo/bar/' f # Zeilen 2-5 sed '2,$s/foo/bar/' f # ab Zeile 2 bis Ende sed '/^#/s/foo/bar/' f # Zeilen die mit # beginnen sed '/START/,/END/d' f # Block löschen sed '0~2s/foo/bar/' f # jede 2. Zeile (GNU sed)
Negation mit !
sed '/^#/!s/foo/bar/' f # alle außer Kommentare sed '2,5!d' f # nur Zeilen 2-5 behalten
🔧
sed — Befehle: d, p, a, i, c, y
Löschen, Drucken, Einfügen
sed '/^#/d' f # Kommentarzeilen löschen sed '/^$/d' f # Leerzeilen löschen sed -n '/ERROR/p' log # nur ERROR-Zeilen ausgeben sed '3a\neue Zeile' f # nach Zeile 3 einfügen sed '3i\neue Zeile' f # vor Zeile 3 einfügen sed '3c\Ersatz' f # Zeile 3 ersetzen sed 'y/abc/ABC/' f # Zeichen transliterieren
Mehrere Befehle & Backreferences
sed -e 's/foo/bar/' -e 's/baz/qux/' sed 's/\(word\)/[\1]/g' # BRE capture group sed -E 's/(word)/[\1]/g' # ERE capture group sed -E 's/([0-9]{4})-([0-9]{2})/\2\/\1/'
💡 sed -n '10,20p' file — schnelles Herausschneiden von Zeilenbereichen
📊
awk — Grundsyntax & Felder
Syntax: awk 'MUSTER {AKTION}' DATEI
awk '{print $1}' f # 1. Feld jeder Zeile awk '{print $NF}' f # letztes Feld awk '{print $(NF-1)}' f # vorletztes Feld awk '{print NR, $0}' f # Zeilennummer + Zeile awk -F: '{print $1}' /etc/passwd # : als Trennzeichen awk -F'[,;]' '{print $2}' f # Regex als FS
Eingebaute Variablen
NR # aktuelle Zeilennummer (total) NF # Anzahl Felder in aktueller Zeile FS # Feldtrennzeichen (default: whitespace) RS # Zeilentrennzeichen (default: \n) OFS # Output-Feldtrenner (default: space) ORS # Output-Zeilentrenner FNR # Zeilennummer in aktueller Datei FILENAME # Name der aktuellen Datei
🔍
awk — Muster & Bedingungen
Musterabgleich & Bedingungen
awk '/ERROR/' log # Zeilen mit ERROR awk '!/^#/' f # ohne Kommentare awk '$3 > 100' f # 3. Feld > 100 awk '$1 == "foo"' f # 1. Feld exakt gleich awk '$2 ~ /[0-9]+/' f # 2. Feld enthält Zahl awk '/START/,/END/' f # Block zwischen START/END awk 'NR==2,NR==5' f # Zeilen 2-5
if/else, Schleifen
awk '{if ($3>100) print "HIGH"; else print "LOW"}' f awk '{for(i=1;i<=NF;i++) print $i}' f awk '{i=1; while(i<=NF){print $i; i++}}' f
🔷
awk — BEGIN, END, Variablen
BEGIN & END Blöcke
awk 'BEGIN{FS=":"; OFS="\t"} {print $1,$3}' /etc/passwd awk 'BEGIN{sum=0} {sum+=$2} END{print "Sum:", sum}' f awk 'END{print NR, "Zeilen"}' f awk 'BEGIN{print "Start"} {print} END{print "Ende"}'
Variablen mit -v übergeben
awk -v threshold=100 '$3 > threshold' f awk -v OFS="," '{print $1,$2,$3}' f
Arrays
awk '{count[$1]++} END{for(k in count) print k, count[k]}' f awk '{sum[$1]+=$2} END{for(k in sum) print k, sum[k]}' f awk '!seen[$0]++' f # Duplikate entfernen
⚡
awk — Eingebaute Funktionen
String-Funktionen
length($0) # Zeilenlänge substr($1, 2, 4) # Teilstring ab Pos 2, Länge 4 split($0, arr, ":") # in Array aufteilen sub(/foo/, "bar", $0) # erstes Vorkommen ersetzen gsub(/foo/, "bar", $0) # alle Vorkommen ersetzen match($0, /[0-9]+/) # Regex-Position (RSTART,RLENGTH) toupper($1) tolower($1) sprintf("%.2f", $3) # formatierter String
Mathematische Funktionen
int($1) sqrt($1) log($1) sin(x) cos(x) atan2(y,x) rand() # 0 ≤ n < 1 srand() # seed für rand()
💡
Nützliche One-Liner
sed One-Liner
# Trailing Whitespace entfernen sed 's/[[:space:]]*$//' f # Leerzeilen entfernen sed '/^[[:space:]]*$/d' f # Zeilennummern hinzufügen sed '=' f | sed 'N;s/\n/\t/' # Erste Zeile entfernen (Header) sed '1d' f # Letzte Zeile entfernen sed '$d' f # Umgekehrte Zeilenfolge sed -n '1!G;h;$p' f
awk One-Liner
# Spalten 1 und 3 mit Tab tauschen awk '{print $3, $2, $1}' f # Zeilen länger als 80 Zeichen awk 'length > 80' f # Summe von Spalte 2 awk '{s+=$2} END{print s}' f # Max-Wert in Spalte 1 awk '$1>max{max=$1} END{print max}' f # Anzahl unique Werte Spalte 1 awk '!a[$1]++ {c++} END{print c}' f # CSV zu Tab-separated awk -F, 'BEGIN{OFS="\t"}{$1=$1; print}' f
Kombination sed + awk
# Config-Wert extrahieren grep "^key" config | awk -F= '{print $2}' | sed 's/ //g' # Log: nur IPs aus Spalte 1 unique awk '{print $1}' access.log | sort | uniq -c | sort -rn
💡 awk -F: '$3==0 {print $1}' /etc/passwd — alle root-UIDs finden

awk — Skript-Dateien & gawk-Extras
awk-Skript in Datei
# report.awk BEGIN { FS=","; print "Report" } $3 > 100 { total += $3; count++ } END { printf "Total: %.2f (%d)\n", total, count } awk -f report.awk data.csv
gawk-spezifische Features
# OFMT für Float-Ausgabe gawk 'BEGIN{OFMT="%.2f"} {print $1+0}' # patsplit: Felder per Regex gawk '{n=patsplit($0,a,/[0-9]+/); print a[1]}' # Mehrere Dateien: BEGINFILE gawk 'BEGINFILE{print "Datei:", FILENAME}' *.log
