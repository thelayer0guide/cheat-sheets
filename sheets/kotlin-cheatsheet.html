Variablen & Typen
val & var
val name: String = "Alice" // unveränderlich var count: Int = 0 // veränderlich val pi = 3.14 // Typinferenz
Null Safety
var s: String? = null // nullable s?.length // safe call s ?: "default" // elvis s!!.length // not-null assert
Typen
Int Long Double Float Boolean Char String Any Unit Nothing
🏗
Klassen & Data Classes
Klasse
class Person( val name: String, var age: Int ) { fun greet() = "Hi, $name" }
data class
data class Point(val x: Int, val y: Int) val p = Point(1, 2) val p2 = p.copy(x = 3) // immutable copy // equals, hashCode, toString generiert
sealed class
sealed class Result<out T> data class Success<T>(val data: T) : Result<T>() data class Error(val msg: String) : Result<Nothing>()
object & companion
object Singleton { fun greet() = "ok" } companion object { fun create() = Person("Alice", 30) }
⚙️
Funktionen & Lambdas
Funktionen
fun add(a: Int, b: Int) = a + b // expression body fun greet(name: String = "Welt"): String { return "Hallo $name" }
Lambda & Higher-Order
val double: (Int) -> Int = { x -> x * 2 } val add = { a: Int, b: Int -> a + b } fun apply(f: (Int) -> Int, x: Int) = f(x) apply(double, 5) // 10 apply({ it * 3 }, 5) // it = einziger Parameter
Extension Functions
fun String.shout() = this.uppercase() + "!" "hello".shout() // "HELLO!"
📦
Collections & Sequences
List / Map / Set
val list = listOf(1, 2, 3) // immutable val ml = mutableListOf(1, 2) val map = mapOf("a" to 1) val mm = mutableMapOf("a" to 1)
Collection-Operationen
list.filter { it > 1 } list.map { it * 2 } list.reduce { acc, x -> acc + x } list.groupBy { it % 2 } list.sortedBy { it } list.first() list.lastOrNull()

when & Smart Casts
when expression
val label = when (x) { 1, 2 -> "eins oder zwei" in 3..9 -> "drei bis neun" is String -> x.uppercase() else -> "andere" }
Smart Cast & is
if (obj is String) { // obj automatisch als String gecasted println(obj.length) }
🧵
Coroutines
Grundlegendes
// build.gradle: kotlinx-coroutines-core import kotlinx.coroutines.* fun main() = runBlocking { val job = launch { delay(1000); println("done") } val res = async { fetchData() } println(res.await()) }
suspend function
suspend fun fetchUser(id: Int): User { return withContext(Dispatchers.IO) { api.getUser(id) } }
Tipp: kotlinx.coroutines.flow für reaktive Streams. Scope immer an Lifecycle binden.

Scope Functions
let / run / apply / also / with
// let: Transformation, null-check user?.let { println(it.name) } // apply: Objekt konfigurieren val sb = StringBuilder().apply { append("hello") append(" world") } // also: Seiteneffekt list.also { println("size: ${it.size}") } // run: Block auf Objekt ausführen user.run { "$name is $age" }
