Partitionierung
Automatisch (clearpart + autopart)
clearpart --all --initlabel autopart --type=lvm # lvm / thinp / plain
Manuell (Standard)
clearpart --all --drives=sda part /boot --fstype=ext4 --size=1024 part /boot/efi --fstype=efi --size=600 part pv.01 --size=1 --grow # PV nimmt Rest volgroup vg_root pv.01 logvol / --vgname=vg_root \ --size=20480 --name=lv_root --fstype=xfs logvol /var --vgname=vg_root \ --size=10240 --name=lv_var logvol swap --vgname=vg_root \ --size=4096 --name=lv_swap
RAID-Partitionen
clearpart --all part raid.01 --size=1024 --ondisk=sda part raid.02 --size=1024 --ondisk=sdb raid /boot --level=1 \ --device=md0 --fstype=ext4 \ raid.01 raid.02

⚡
%pre & %post Skripte
%pre — Vor der Installation
%pre #!/bin/bash # Läuft im Installer-RAM-Disk # Disk bestimmen, Netzwerk prüfen, etc. DISK=$(lsblk -d -n -o NAME | head -1) echo "ignoredisk --only-use=$DISK" \ > /tmp/part-include %end
%post — Nach der Installation
%post --log=/root/ks-post.log #!/bin/bash # Läuft im installierten System # SSH-Key deployen: mkdir -p /root/.ssh echo "ssh-ed25519 AAAA... deploy" \ >> /root/.ssh/authorized_keys chmod 700 /root/.ssh chmod 600 /root/.ssh/authorized_keys # Pakete nachinstallieren: dnf -y install ansible-core # Erster Ansible-Run: ansible-pull -U https://git.example.com/configs.git %end
%post ohne chroot
%post --nochroot # Zugriff auf Installer-Dateisystem cp /run/install/repo/custom.conf /mnt/sysimage/etc/ %end

PXE-Boot / Deployment
Kernel-Boot-Parameter
# PXE-Kernel-Cmdline: vmlinuz initrd=initrd.img \ inst.ks=http://192.168.1.10/ks.cfg \ inst.repo=http://192.168.1.10/rhel9 # NFS: inst.ks=nfs:192.168.1.10:/ks/ks.cfg # USB / Lokal: inst.ks=hd:sdb1:/ks.cfg
Einfacher HTTP-Server für KS-File
# Python-Webserver starten: python3 -m http.server 8080 --directory /srv/ks # Oder Nginx: cp ks.cfg /usr/share/nginx/html/ # URL: http://server:80/ks.cfg # SELinux-Kontext bei httpd: chcon -t httpd_sys_content_t /srv/ks/ks.cfg
