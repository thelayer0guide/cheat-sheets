Grundstruktur
Minimales Programm
#include <stdio.h> int main(int argc, char *argv[]) { printf("Hallo\n"); return 0; }
Kompilieren & Ausführen
gcc -Wall -Wextra -O2 main.c -o main ./main
Mit Debuginfo
gcc -g -fsanitize=address main.c -o main

Datentypen
Ganzzahlen

float f = 3.14f; double d = 3.141592;

Kontrollstrukturen
if / else
if (x > 0) { printf("positiv\n"); } else if (x == 0) { printf("null\n"); } else { printf("negativ\n"); }
for / while / do-while
for (int i=0; i<10; i++) { ... } while (cond) { ... } do { ... } while (cond);
switch
switch (c) { case 'a': ...; break; default: ...; break; }

Arrays & Strings
Array
int arr[5] = {1,2,3,4,5}; int len = sizeof(arr)/sizeof(arr[0]);

char s[] = "hello"; // 6 Byte char buf[64]; snprintf(buf, sizeof(buf), "%s!", s);
string.h Funktionen
strlen(s) // Länge strcpy(dst, src) // Kopieren strncpy(dst,src,n) strcmp(a, b) // Vergleich strcat(dst, src) // Anhängen strchr(s, 'c') // Suchen
🎯
Pointer
Grundlagen
int x = 42; int *p = &x; // Adresse printf("%d", *p); // Dereferenz *p = 99; // Schreiben
Pointer-Arithmetik
int arr[] = {1,2,3}; int *p = arr; p++; // nächstes Element *(p+1) == arr[2];
NULL-Check
if (p == NULL) { /* Fehler */ } int *p = NULL; // sicher init
Funktionszeiger
int (*fp)(int,int) = add; int r = fp(3, 4);
🧠
Dynamischer Speicher
malloc / calloc / realloc / free
#include <stdlib.h> int *p = malloc(10 * sizeof(int)); int *q = calloc(10, sizeof(int)); p = realloc(p, 20 * sizeof(int)); free(p); p = NULL;
Prüfung
if (!p) { perror("malloc"); exit(1); }
Regel: Jedes malloc braucht ein free. Niemals doppelt free().

Structs & Enums
struct
typedef struct { char name[64]; int age; } Person; Person p = {"Alice", 30}; Person *ptr = &p; ptr->age = 31;
enum
typedef enum { RED, GREEN, BLUE } Color; Color c = GREEN;
union
union Data { int i; float f; }; union Data d; d.i = 42;
⚙️
Funktionen
Deklaration & Definition
int add(int a, int b); // Prototyp int add(int a, int b) { return a + b; }
Pass by pointer
void swap(int *a, int *b) { int t = *a; *a = *b; *b = t; } swap(&x, &y);
Variadic (stdarg.h)
#include <stdarg.h> int sum(int n, ...) { va_list ap; va_start(ap, n); for(int i=0;i<n;i++) va_arg(ap,int); va_end(ap); }
📁
Datei-I/O
Öffnen / Schließen
FILE *f = fopen("data.txt", "r"); if (!f) { perror("fopen"); } fclose(f);
Modi
"r" Lesen "w" Schreiben "a" Anhängen "rb" Binär lesen "r+" Lesen+Schreiben
Lesen / Schreiben
fscanf(f, "%d", &n); fprintf(f, "%s\n", s); fgets(buf, sizeof(buf), f); fwrite(data, sizeof(*data), n, f);

Präprozessor
Makros
#define MAX(a,b) ((a)>(b)?(a):(b)) #define ARRAY_LEN(a) (sizeof(a)/sizeof(a[0]))
Include Guard
#ifndef MYHEADER_H #define MYHEADER_H /* Inhalt */ #endif
Bedingte Kompilierung

Nützliche Makros

⚠️
Häufige Fehler
Buffer Overflow
// FALSCH: gets(buf); // RICHTIG: fgets(buf, sizeof(buf), stdin);
Use-After-Free
free(p); p = NULL; // IMMER auf NULL setzen
Signed/Unsigned Vergleich
size_t len = strlen(s); for (size_t i=0; i<len; i++) // korrekt
Tipp: Immer mit -Wall -Wextra kompilieren und Valgrind/ASan nutzen.
