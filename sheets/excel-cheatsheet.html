Formeln — Lookup
XLOOKUP (modern)
=XLOOKUP(Suchwert, Suchbereich, Rückgabebereich, "nicht gef.") =XLOOKUP(A2, C:C, D:D) einfach =XLOOKUP(A2, C:C, D:D, "-", 0) exakt
VLOOKUP / HLOOKUP
=VLOOKUP(A2, C:E, 2, 0) -- Wert, Tabelle, Spaltenindex, [exakt] =HLOOKUP(A2, 1:3, 2, 0)
INDEX / MATCH
=INDEX(D:D, MATCH(A2, C:C, 0)) -- flexibler als VLOOKUP =INDEX(B2:B100, MATCH(1,(A2:A100=X)*(C2:C100=Y),0))

Formeln — Logik & Statistik
Bedingte Funktionen
=IF(A2>100, "hoch", "niedrig") =IFS(A2>90,"A", A2>70,"B", TRUE,"C") =IFERROR(A2/B2, 0) =SWITCH(A2, 1,"Jan", 2,"Feb", "?")
Aggregation mit Bedingung
=SUMIF(B:B, "Ja", C:C) =SUMIFS(C:C, A:A, "DE", B:B, ">2023") -- B = Jahreszahl =SUMIFS(C:C, A:A, "DE", D:D, ">="&DATE(2024,1,1)) -- D = Datum =COUNTIF(A:A, ">0") =COUNTIFS(A:A, ">0", B:B, "Aktiv") =AVERAGEIF(B:B, "<>0", C:C)
Statistik & Aggregation
=SUM(A1:A100) =AVERAGE(A:A) =MIN(A:A) =MAX(A:A) =MEDIAN(A:A) =STDEV(A:A) =COUNTA(A:A) -- nicht-leere Zellen =SUBTOTAL(9,A:A) -- nur sichtbare
