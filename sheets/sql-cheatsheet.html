📑
Data Manipulation (DML)
SELECT - Daten abfragen
SELECT id, name FROM users; SELECT * FROM users; -- Alle Spalten SELECT DISTINCT role FROM users; -- Duplikate entfernen
INSERT - Zeilen hinzufügen
INSERT INTO users (name, age) VALUES ('Alice', 30), ('Bob', 25);
UPDATE - Daten ändern
UPDATE users SET age = 31, active = true WHERE name = 'Alice';
DELETE - Zeilen löschen
DELETE FROM users WHERE id = 5;
🏗️
Data Definition (DDL)
CREATE - Tabellen erstellen
CREATE TABLE users ( id SERIAL PRIMARY KEY, name VARCHAR(50) NOT NULL, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP );
ALTER - Tabellenstruktur ändern
ALTER TABLE users ADD COLUMN age INT; ALTER TABLE users DROP COLUMN age; ALTER TABLE users RENAME TO customers;
DROP - Tabellen/Datenbanken löschen
DROP TABLE users; DROP TABLE IF EXISTS users; -- Ohne Fehler, falls nicht existent DROP DATABASE my_db;
TRUNCATE - Tabelle leeren
TRUNCATE TABLE users; -- Schneller als DELETE, resettet IDs
🔍
Filter & Joins
WHERE - Bedingungen
SELECT * FROM users WHERE age >= 18 AND status = 'active';
LIKE - Mustererkennung (Pattern Matching)
SELECT * FROM users WHERE name LIKE 'A%'; -- Beginnt mit A -- % = beliebig viele Zeichen, _ = genau ein Zeichen
IN & BETWEEN
SELECT * FROM users WHERE role IN ('admin', 'editor'); SELECT * FROM users WHERE age BETWEEN 20 AND 30;
INNER JOIN - Schnittmenge
SELECT u.name, o.total FROM users u INNER JOIN orders o ON u.id = o.user_id;
LEFT JOIN - Alle von links & Matches von rechts
SELECT u.name, o.total FROM users u LEFT JOIN orders o ON u.id = o.user_id;
🧮
Aggregation & Sortierung
ORDER BY & LIMIT
SELECT * FROM users ORDER BY created_at DESC -- DESC absteigend, ASC aufsteigend LIMIT 10 OFFSET 20; -- Paginierung: Zeige Ergebnisse 21-30
Aggregatfunktionen
SELECT COUNT(*) FROM users; -- Anzahl Zeilen SELECT SUM(total), AVG(total) FROM orders; -- Summe & Durchschnitt SELECT MIN(age), MAX(age) FROM users;
GROUP BY & HAVING
SELECT role, COUNT(*) AS user_count FROM users GROUP BY role HAVING COUNT(*) > 5; -- HAVING filtert nach Aggregation
💡 WHERE filtert vor dem Gruppieren, HAVING filtert nach dem Gruppieren.

Transaktionen & Indizes
Transaktionen (ACID)
BEGIN; -- Oder START TRANSACTION; UPDATE accounts SET balance = balance - 100 WHERE id = 1; UPDATE accounts SET balance = balance + 100 WHERE id = 2; COMMIT; -- Bestätigen -- ROLLBACK; -- Falls Fehler, alles rückgängig machen
Indizes (Performance)
CREATE INDEX idx_user_email ON users (email); CREATE UNIQUE INDEX idx_user_username ON users (username); DROP INDEX idx_user_email;
🔗
Alle JOIN-Typen
RIGHT JOIN - alle von rechts + matches links
SELECT u.name, o.total FROM users u RIGHT JOIN orders o ON u.id = o.user_id;
FULL OUTER JOIN - alle Zeilen beider Seiten
SELECT u.name, o.total FROM users u FULL OUTER JOIN orders o ON u.id = o.user_id;
CROSS JOIN - kartesisches Produkt
SELECT a.color, b.size FROM colors a CROSS JOIN sizes b;
Self JOIN - Tabelle mit sich selbst
SELECT e.name, m.name AS manager FROM employees e LEFT JOIN employees m ON e.manager_id = m.id;
💡 MySQL kennt kein FULL OUTER JOIN — stattdessen: LEFT JOIN UNION RIGHT JOIN
🔀
NULL & CASE
NULL prüfen
SELECT * FROM users WHERE email IS NULL; SELECT * FROM users WHERE email IS NOT NULL;
COALESCE & NULLIF
COALESCE(email, 'n/a') -- erster Nicht-NULL-Wert NULLIF(score, 0) -- NULL wenn score = 0
CASE WHEN
SELECT name, CASE WHEN age < 18 THEN 'Junior' WHEN age < 65 THEN 'Erwachsen' ELSE 'Senior' END AS gruppe FROM users;
CASE in Aggregation (Pivot)
SELECT SUM(CASE WHEN status='active' THEN 1 ELSE 0 END) AS aktive, SUM(CASE WHEN status='inactive' THEN 1 ELSE 0 END) AS inaktive FROM users;
⛙
Mengenoperationen
UNION — vereinigt, Duplikate entfernt
SELECT name FROM customers UNION SELECT name FROM employees;
UNION ALL — mit Duplikaten (schneller)
SELECT id, 'order' AS type FROM orders UNION ALL SELECT id, 'return' AS type FROM returns;
INTERSECT & EXCEPT
SELECT name FROM t1 INTERSECT SELECT name FROM t2; -- Schnittmenge (in beiden) SELECT name FROM t1 EXCEPT SELECT name FROM t2; -- Differenz (nur in t1)
💡 Alle Mengenoperationen erfordern gleiche Spaltenanzahl und kompatible Typen
🔎
Subqueries & CTEs
Subquery in WHERE
SELECT * FROM users WHERE id IN ( SELECT user_id FROM orders WHERE total > 100 );
EXISTS / NOT EXISTS
SELECT * FROM users u WHERE EXISTS ( SELECT 1 FROM orders o WHERE o.user_id = u.id ); -- NOT EXISTS: Nutzer ohne Bestellungen
Skalare Subquery in SELECT
SELECT name, (SELECT COUNT(*) FROM orders WHERE user_id = u.id) AS anzahl_bestellungen FROM users u;
CTE — WITH ... AS
WITH top_users AS ( SELECT user_id, SUM(total) AS umsatz FROM orders GROUP BY user_id HAVING SUM(total) > 500 ) SELECT u.name, t.umsatz FROM top_users t JOIN users u ON u.id = t.user_id;
💡 Mehrere CTEs: WITH cte1 AS (...), cte2 AS (...) SELECT ... — für lesbarere Queries
🪟
Window-Funktionen (OVER)
Grundsyntax
funktion() OVER ( [PARTITION BY spalte] [ORDER BY spalte ASC|DESC] )
Ranking-Funktionen
SELECT name, salary, ROW_NUMBER() OVER (ORDER BY salary DESC) AS zeile, RANK() OVER (ORDER BY salary DESC) AS rang, DENSE_RANK() OVER (ORDER BY salary DESC) AS dichte FROM employees;
PARTITION BY — Rang pro Gruppe
SELECT dept, name, salary, RANK() OVER ( PARTITION BY dept ORDER BY salary DESC ) AS rang_im_dept FROM employees;
LAG & LEAD — Vor-/Nachzeile
SELECT date, revenue, LAG(revenue) OVER (ORDER BY date) AS vorher, LEAD(revenue) OVER (ORDER BY date) AS danach FROM sales;
Laufende Summe / Durchschnitt
SELECT name, salary, SUM(salary) OVER (PARTITION BY dept) AS dept_sum, AVG(salary) OVER (ORDER BY date ROWS BETWEEN 2 PRECEDING AND CURRENT ROW) AS moving_avg FROM employees;
🏛
Views & Constraints
Views (virtuelle Tabellen)
CREATE VIEW aktive_nutzer AS SELECT id, name, email FROM users WHERE active = true; SELECT * FROM aktive_nutzer; DROP VIEW aktive_nutzer;
Constraints beim CREATE TABLE
CREATE TABLE orders ( id SERIAL PRIMARY KEY, user_id INT NOT NULL REFERENCES users(id) ON DELETE CASCADE, total NUMERIC(10,2) CHECK (total >= 0), status VARCHAR(20) DEFAULT 'pending', code VARCHAR(20) UNIQUE );
Constraint nachträglich hinzufügen
ALTER TABLE orders ADD CONSTRAINT fk_user FOREIGN KEY (user_id) REFERENCES users(id);
