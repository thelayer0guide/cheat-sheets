crictl — Kubernetes CRI
Konfiguration
cat > /etc/crictl.yaml <
Container
crictl ps # laufende Container crictl ps -a # alle Container crictl exec -it sh crictl logs crictl logs -f crictl inspect crictl stats # Ressourcen crictl stop crictl rm
Pods & Images
crictl pods crictl pods --name myapp crictl images crictl rmi crictl pull nginx:alpine
🐳
nerdctl — Docker-Kompatibel
Container verwalten
nerdctl run -d --name nginx nginx:alpine nerdctl ps nerdctl ps -a nerdctl stop nginx nerdctl rm nginx nerdctl exec -it nginx sh nerdctl logs -f nginx
Images
nerdctl images nerdctl pull ubuntu:22.04 nerdctl push myregistry/myapp:v1 nerdctl build -t myapp:latest . nerdctl tag myapp:latest myregistry/myapp:latest
Namespaces
nerdctl --namespace k8s.io ps # K8s-Namespace nerdctl namespace ls
nerdctl ist ein Docker-kompatibler CLI für containerd. Bei Kubernetes-Debugging ist crictl die richtige Wahl.
⚙️
ctr — Low-Level CLI
Images
ctr images ls ctr images pull docker.io/library/nginx:alpine ctr images push myregistry/myapp:v1 ctr images rm nginx:alpine
Container
ctr containers ls ctr containers create docker.io/library/nginx:alpine nginx ctr tasks start nginx ctr tasks ls ctr tasks exec --exec-id sh nginx sh ctr tasks kill nginx ctr tasks rm nginx ctr containers rm nginx
Namespace-Flag
# Kubernetes verwendet den Namespace "k8s.io": ctr -n k8s.io containers ls ctr -n k8s.io images ls
🛠
containerd Service
Service verwalten
systemctl status containerd systemctl restart containerd journalctl -u containerd -f
/etc/containerd/config.toml
version = 2 [plugins."io.containerd.grpc.v1.cri"] [plugins."io.containerd.grpc.v1.cri".containerd] [plugins."io.containerd.grpc.v1.cri".containerd.runtimes.runc] runtime_type = "io.containerd.runc.v2" [plugins."io.containerd.grpc.v1.cri".registry] [plugins."io.containerd.grpc.v1.cri".registry.mirrors] [plugins."io.containerd.grpc.v1.cri".registry.mirrors."docker.io"] endpoint = ["https://mirror.gcr.io"]
Konfig generieren & prüfen
containerd config default > /etc/containerd/config.toml containerd config dump
