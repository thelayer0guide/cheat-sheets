apt — Pakete verwalten
Paketlisten aktualisieren & upgraden
apt update apt upgrade apt full-upgrade # dist-upgrade (Abhängigkeiten lösen) apt dist-upgrade # Alias für full-upgrade
Pakete installieren / entfernen
apt install nginx apt install -y nginx curl wget # ohne Nachfrage apt remove nginx # Paket, Konfig bleibt apt purge nginx # Paket + Konfig apt autoremove # verwaiste Abhängigkeiten apt install --reinstall nginx
Suche & Informationen
apt search nginx apt show nginx apt list --installed apt list --upgradable apt-cache depends nginx apt-cache rdepends nginx # wer braucht nginx?
💡 apt-get für Skripte bevorzugen (stabile Ausgabe), apt für interaktive Nutzung
🔧
dpkg — Low-Level
Pakete auflisten & abfragen
dpkg -l # alle installierten Pakete dpkg -l nginx* dpkg -L nginx # Dateien des Pakets dpkg -S /usr/sbin/nginx # welches Paket? (file → pkg) dpkg -s nginx # Paket-Status
Paket installieren / entfernen
dpkg -i paket.deb # .deb installieren dpkg -r nginx # entfernen (Konfig bleibt) dpkg -P nginx # purge dpkg --configure -a # unterbrochene Konfigurationen reparieren
🗄
apt — Repositories & Schlüssel
Quellen verwalten
# /etc/apt/sources.list deb https://deb.debian.org/debian bookworm main deb-src https://deb.debian.org/debian bookworm main # Einzeldatei in /etc/apt/sources.list.d/ add-apt-repository "deb https://pkg.example.com bookworm main" add-apt-repository --remove "deb ..."
GPG-Schlüssel (moderner Weg)
curl -fsSL https://pkg.example.com/key.gpg \ | gpg --dearmor -o /etc/apt/keyrings/example.gpg # sources.list mit signed-by deb [signed-by=/etc/apt/keyrings/example.gpg] \ https://pkg.example.com bookworm main
apt-mark — Pakete fixieren
apt-mark hold nginx # Version einfrieren apt-mark unhold nginx apt-mark showhold # alle gehaltenen Pakete apt-mark manual nginx # als manuell markieren
Pinning (/etc/apt/preferences)
Package: nginx Pin: release a=stable Pin-Priority: 1001 # >1000 = erzwingen Package: * Pin: release o=Ubuntu Pin-Priority: 500 # Standard
💡 apt-cache policy nginx — zeigt verfügbare Versionen und Prioritäten

dnf / yum — RHEL, Fedora, CentOS
Pakete verwalten
dnf check-update dnf upgrade dnf install nginx dnf remove nginx dnf autoremove dnf reinstall nginx
Suche & Informationen
dnf search nginx dnf info nginx dnf list --installed dnf provides /usr/sbin/nginx # welches Paket liefert Datei? dnf repoquery -l nginx # Dateien des Pakets
Repositories
dnf repolist dnf repolist --all dnf config-manager --enable epel dnf config-manager --disable epel dnf config-manager --add-repo https://repo.example.com/el9.repo
Gruppen & Historie
dnf group list dnf group install "Development Tools" dnf history dnf history undo 5 # Transaktion 5 rückgängig dnf history info 5
💡 yum ist unter RHEL 8+ ein Alias für dnf — gleiche Syntax
🔵
rpm — Low-Level (RHEL/Fedora)
Abfragen
rpm -qa # alle installierten Pakete rpm -qi nginx # Paket-Info rpm -ql nginx # Dateien des Pakets rpm -qf /usr/sbin/nginx # welches Paket? rpm -qR nginx # Abhängigkeiten
Installieren / Entfernen
rpm -Uvh paket.rpm # upgrade (auch neu-install) rpm -ivh paket.rpm # nur install rpm -e nginx # entfernen rpm -V nginx # Paket verifizieren
🦎
zypper — openSUSE / SLES
Pakete verwalten
zypper refresh # Repos aktualisieren zypper update zypper dist-upgrade # du zypper install nginx # in zypper remove nginx # rm / remove zypper clean # Cache leeren
Suche & Repos
zypper search nginx # se zypper info nginx zypper repos # lr zypper addrepo URL NAME # ar zypper removerepo NAME # rr zypper modifyrepo --enable 1
Locks (Pinning)
zypper addlock nginx zypper removelock nginx zypper locks
🔷
pacman — Arch Linux
Pakete verwalten
pacman -Syu # sync + update (immer beides!) pacman -S nginx # installieren pacman -R nginx # entfernen pacman -Rns nginx # + Abhängigkeiten + Konfig pacman -Sc # Cache bereinigen
Suche & Abfragen
pacman -Ss nginx # suchen pacman -Si nginx # Info (remote) pacman -Qi nginx # Info (installiert) pacman -Ql nginx # Dateien des Pakets pacman -Qo /usr/sbin/nginx # welches Paket? pacman -Qdt # verwaiste Pakete
💡 AUR-Helper: yay -Syu oder paru -Syu (auch AUR-Pakete updaten)

snap & flatpak
snap (Canonical)
snap find vscode snap install code --classic snap list snap refresh # alle aktualisieren snap refresh code snap remove code snap info code
flatpak
flatpak remote-add --if-not-exists \ flathub https://flathub.org/repo/flathub.flatpakrepo flatpak search firefox flatpak install flathub org.mozilla.firefox flatpak list flatpak update flatpak uninstall org.mozilla.firefox
⚡
Nützliche Einzeiler
Welches Paket liefert eine Datei? (apt)
apt-file search /usr/bin/netstat dpkg -S /usr/bin/netstat # nur installierte Pakete
Größte installierte Pakete
dpkg-query -W -f '${Installed-Size}\t${Package}\n' \ | sort -rn | head -20
Sicherheitsupdates (apt)
apt -s upgrade | grep security unattended-upgrades --dry-run
Paket-Cache bereinigen
apt clean # heruntergeladene .deb löschen apt autoclean # nur veraltete .deb löschen dnf clean all pacman -Sc
Nicht verwaltete Pakete finden
deborphan # Debian: verwaiste Pakete pacman -Qdt # Arch: nicht benötigte Pakete dnf leaves # RHEL: Blätter-Pakete (kein Dependent)
💡 needrestart nach apt upgrade — zeigt Dienste die neu gestartet werden müssen
