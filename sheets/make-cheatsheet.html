Makefile — Grundstruktur
Grundlegende Syntax
# Kommentar # target: prerequisites # command (TAB, kein Leerzeichen!) all: build test build: main.c utils.c gcc -o app main.c utils.c clean: rm -f app *.o .PHONY: all build clean test
make — Aufruf
make # Standard-Target make build # Spezifisches Target make -n # Dry-run (nur anzeigen) make -j4 # 4 parallele Jobs make -f other.mk # anderes Makefile make -B # Immer neu bauen make -k # Weitermachen bei Fehler make VAR=value # Variable übergeben

DevOps-Makefile Muster
Standard-DevOps-Makefile
.DEFAULT_GOAL := help .PHONY: help build push deploy clean IMAGE := registry.example.com/myapp TAG := $(shell git rev-parse --short HEAD) help: @grep -E '^[a-zA-Z_-]+:.*?##' $(MAKEFILE_LIST) | \ awk 'BEGIN{FS=":.*?## "}{printf "%-15s %s\n",$$1,$$2}' build: ## Docker-Image bauen docker build -t $(IMAGE):$(TAG) . push: build ## Image in Registry pushen docker push $(IMAGE):$(TAG) deploy: push ## In Kubernetes deployen kubectl set image deploy/myapp app=$(IMAGE):$(TAG)
Stille Ausgabe & Fehlerbehandlung
clean: @rm -rf dist/ # @ = kein Echo -rm -f *.log # - = Fehler ignorieren install: $(if $(shell which curl),,$(error curl not found)) @echo "Installing..."
