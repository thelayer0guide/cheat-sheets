GitLab CI — Grundstruktur
.gitlab-ci.yml — Minimal-Beispiel
stages: - build - test - deploy variables: IMAGE: registry.example.com/myapp APP_ENV: staging build-job: stage: build image: docker:24 script: - docker build -t $IMAGE:$CI_COMMIT_SHA . - docker push $IMAGE:$CI_COMMIT_SHA
Vordefinierte Variablen
$CI_COMMIT_SHA # Commit-Hash $CI_COMMIT_SHORT_SHA # 8-stellig $CI_COMMIT_BRANCH # Branch-Name $CI_COMMIT_TAG # Tag-Name $CI_PIPELINE_ID # Pipeline-ID $CI_PROJECT_NAME # Projekt-Name $CI_REGISTRY # Registry-URL $CI_REGISTRY_USER # Registry-User $CI_REGISTRY_PASSWORD # Registry-Pass
🎯
Jobs — Optionen
Job-Konfiguration
deploy-prod: stage: deploy image: alpine:3.18 environment: name: production url: https://myapp.example.com only: - main when: manual needs: [build-job, test-job] tags: - production-runner allow_failure: false
rules: (empfohlen statt only/except)
rules: - if: '$CI_COMMIT_BRANCH == "main"' when: always - if: '$CI_COMMIT_TAG =~ /^v\d+/' when: always - when: never # Default: nie # Nur wenn Datei geändert: - changes: - src/**/* - Dockerfile

GitHub Actions
.github/workflows/ci.yml — Grundstruktur
name: CI on: push: branches: [main] pull_request: branches: [main] workflow_dispatch: jobs: build: runs-on: ubuntu-latest steps: - uses: actions/checkout@v4 - name: Setup Node uses: actions/setup-node@v4 with: node-version: '20' - run: npm ci - run: npm test
GitHub Actions — Umgebungen & Secrets
# Secrets aus GitHub: env: TOKEN: ${{ secrets.DEPLOY_TOKEN }} ENV: ${{ vars.APP_ENV }} # Environment mit Approval: jobs: deploy: environment: name: production url: https://myapp.com runs-on: ubuntu-latest
